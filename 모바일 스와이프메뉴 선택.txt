var xcode = (function(n) {
    n = n.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
    var r = new RegExp("[\\?&]" + n + "=([^&#]*)"),
        s = r.exec(location.search);
    return s == null ? "" : decodeURIComponent(s[1].replace(/\+/g, " "));
})('xcode') || '';
