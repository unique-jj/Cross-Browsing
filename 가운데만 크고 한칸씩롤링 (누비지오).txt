function mdpickSlideInit() {
	var circleSlide = jQuery("#mdpick_slide");  //슬라이드 객체
	var getSlideWidth = circleSlide.width();
	var slideAuto = true;   //자동
	var slideControl = false;    //좌우 화살표 사용
	var slideHover = true;      //마우스 오버 자동 멈춤
	var slidePause = 4000;  //다음 슬라이드 시간
	var slideSpeed = 600;   //슬라이드 에니메이션 속도
	var slideEasing = "easeOutQuad";    //슬라이드 easing 효과
	var slideWidth = 280;   //슬라이드 이미지 크기
	var slideTop = 100;     //슬라이드 상단 여백
	var slideOpacity = 1; //슬라이드 반투명 값
	var slideMargin = 50;   //슬라이드 좌우 여백 
	var slideCenterMargin = 38; //슬라이드 가운데 여백
	
	var slideIdx = 5;
	var slideScale = 0.5;
	var slideCenterExtraMargin = (slideWidth - (slideWidth * slideScale)) / 2;
	var slideCenterLeft = (getSlideWidth - (((slideIdx - 1) * (slideWidth / 2 + slideMargin)) + (slideWidth + slideCenterMargin))) / 2;
	var slideRunCheck = true;
	circleSlide.append(circleSlide.html());
	
	var circleSlideInterval = null;
	function circleSlideRun(type, target, dir) {
		if(!slideRunCheck) return false;
		clearTimeout(circleSlideInterval);
		slideRunCheck = false;
		var getAnimate = type;
		if(type && dir == "prev") {
			var getPrev = jQuery(target).children().last();
			jQuery(target).prepend(getPrev);
			getPrev.css("left",-1000);
		}
		else if(type && dir == "next") {
			var getPrev = jQuery(target).children().first();
			jQuery(target).append(getPrev);
			getPrev.css("left",10000);
		}
		jQuery(target).children("li").each(function(i) {
			var getIdx = i - 2;
			var getWidth = slideWidth * slideScale;
			var getHalf = getWidth / 2;
			var getMargin = slideMargin;
			var getExtraMargin = 0;
			var getImgWidth = getWidth;
			var getOpacity = slideOpacity;
			if(getIdx == 2) {
				getHalf = slideWidth / 2;
				getImgWidth = slideWidth;
				getExtraMargin = (slideCenterMargin / 2) + slideCenterExtraMargin;
				getOpacity = 1;
			}
			else if(getIdx > 2) {
				getExtraMargin = getHalf + slideCenterMargin + slideCenterExtraMargin;
			}
			
			var liStyle = { width:getWidth, height:getWidth, top:slideTop, left:(getIdx * getWidth) + (getIdx * getMargin + getExtraMargin) + slideCenterLeft, opacity:getOpacity };
			var imgStyle = { width:getImgWidth, height:getImgWidth, marginLeft:-getHalf, marginTop:-getHalf };
			if(getAnimate) {
				jQuery(this).stop(true).animate(liStyle, slideSpeed, slideEasing).find("img").stop(true).animate(imgStyle, slideSpeed, slideEasing, function() {
					slideRunCheck = true;
				});
			}
			else {
				jQuery(this).css(liStyle).find("img").css(imgStyle);
			}
		});
		
		if(!getAnimate) {
			slideRunCheck = true;
			target.addClass("active");
			if(slideControl) jQuery(".mdpick_slide_box .btn_slide").addClass("active");
		}
		circleSlideAuto();
	}
	function circleSlideAuto() {
		if(slideAuto) {
			circleSlideInterval = setTimeout(function() {
				circleSlideRun(true, circleSlide, "next");
			}, slidePause);
		}
	}
	circleSlideRun(false, circleSlide);
	
	if(slideAuto && slideHover) {
		circleSlide.find("li").hover(
			function() {
				clearTimeout(circleSlideInterval);
			},
			function() {
				circleSlideAuto();
			}
		);
		jQuery(".mdpick_slide_box .btn_slide").hover(
			function() {
				clearTimeout(circleSlideInterval);
			},
			function() {
				circleSlideAuto();
			}
		);
	}
	
	jQuery(".mdpick_slide_box .btn_slide.prev").on("click", function() {
		circleSlideRun(true, circleSlide, "prev");
		return false;
	});
	
	jQuery(".mdpick_slide_box .btn_slide.next").on("click", function() {
		circleSlideRun(true, circleSlide, "next");
		return false;
	});

}
