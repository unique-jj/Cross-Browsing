<script type="text/javascript">
//menu show/hide script
(function($) {
    //img.src on/off 처리 function
    $.fn._img_on = function() {
        return this.each(function() {
            $(this).attr('src', $(this).attr('src').replace('_off', '_on'));
        });
    };
    $.fn._img_off = function() {
        return this.each(function() {
            $(this).attr('src', $(this).attr('src').replace('_on', '_off'));
        });
    };
    
    $('.brand-menu li a').each(function(i) {
        var _th = $(this);
        _th.on('click', function(e) {
			e.preventDefault(); //a 태그 작동하지 말아라
            $('.brand-menu li a img')._img_off().eq(i)._img_on();    //싹다 off 하고 해당 놈만 on
            $('.brand-sub').hide().eq(i).show();   //싹다 hide 하고 해당 놈만 show
        });
    }).eq(0).trigger('click');    
})(jQuery);
</script>
