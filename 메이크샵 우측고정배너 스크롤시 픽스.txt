<script>
$(function(){
  $(window).scroll(function(){
    if($(this).scrollTop() >200){
      $("#blk_scroll_wings").addClass("topfixed");
    }else{
      $("#blk_scroll_wings").removeClass("topfixed"); 
    }
    if($(this).scrollTop() >890){
      $("#blk_scroll_wings").removeClass("topfixed");
      $("#blk_scroll_wings").addClass("fixed");
    }else{
      $("#blk_scroll_wings").removeClass("fixed"); 
    }
  });
});
</script>
